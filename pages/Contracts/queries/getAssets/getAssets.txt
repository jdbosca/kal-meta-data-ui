FROM VIEW_DB.DATA_SCHEMA.ASSETS_V

    ({{asset_type_select.selectedOptionValue ? 'ASSET_TYPE = ' + asset_type_select.selectedOptionValue : '1=1'}}) AND
    ({{minesite_select.selectedOptionValue ? 'MINE_REF = ' + minesite_select.selectedOptionValue : '1=1'}}) AND
    ({{parent_account_select.selectedOptionValue ? 'PARENT_ACCOUNT_ID = ' + parent_account_select.selectedOptionValue : '1=1'}});
